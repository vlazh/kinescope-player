Пример использования

const id = 'playerContainerId';

/** @type {PlayerOptions} */
const options = {
  playlist: [
    {
      sources: {
        hls: 'path-to/master.m3u8',
        // or mp4 format
        mp4: {
          '360': 'path-to/360.mp4',
          '480': 'path-to/480.mp4',
          '720': 'path-to/720.mp4',
          '1080': 'path-to/1080.mp4',
          '2160': 'path-to/2160.mp4',
        },
      },

      poster: 'path-to/poster.jpg',
      title: 'TITLE OF VIDEO',
      subtitle: 'SUBTITLE OF VIDEO',

      vtt: [
        {
          label: 'Русский',
          src: 'path-to/subtitles_ru.vtt',
          srcLang: 'ru',
        },
        {
          label: 'English',
          src: 'path-to/subtitles_en.vtt',
          srcLang: 'en',
        },
      ],

      chapters: [
        { position: 0, title: 'Chapter 1' },
        { position: 50, title: 'Chapter 2' },
        { position: 150, title: 'Chapter 3' },
        { position: 260, title: 'Chapter 4' },
        { position: 320, title: 'Chapter 5' },
        { position: 400, title: 'Chapter 6' },
        { position: 450, title: 'Chapter 7' },
        { position: 510, title: 'Chapter 8' },
        { position: 600, title: 'Chapter 9' },
        { position: 680, title: 'Chapter 10' },
        { position: 710, title: 'Chapter 11' },
      ],
    },
  ],

  size: {
    width: '100%',
    height: '100%',
  },

  behavior: {
    preload: 'none',
  },

  ui: {
    language: 'en',
    playbackRateButton: true,
    openChaptersOnLoad: true,
  },

  files: {
    list: [
      {
        name: 'File name here',
        url: 'path-to/file1.jpg',
        size: 123,
        mime: 'application/zip',
      },
      {
        name: 'File name here',
        url: 'path-to/file2.jpg',
        size: 1234,
        mime: 'application/msword',
      },
    ],

    archiveUrl: 'path-to/all.zip',
  },
};

/** @type {PlayerController} */
const player = await Kinescope.Player.create(id, options);
// or
// const player = Kinescope.Player.getById(id);
player.setPlaylistItemOptions({ title: 'New title' });
