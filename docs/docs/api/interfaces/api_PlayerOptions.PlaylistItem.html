            id?: string;

        | Omit<PopupCTAItem, "id" | "type"> & { id?: string; type?: "popup" }
        | Omit<ButtonsCTAItem, "id" | "type"> & { id?: string; type?: "buttons" }

    )[];
    downloads?: {

        list?: readonly { name?: string; size?: number; url?: string }[];
    };
    drm?: DRMOptions;
    files?: {
        archiveUrl?: string;
        list?: readonly { name?: string; size?: number; url?: string }[];
    };
    frameRate?: number
    | { [P: number]: number };
    heatmap?: string | readonly Point[];
    id?: string;
    liveRevivalCheck?: Pick<LiveCheckerOptions, "url" | "interval">;
    liveStartsAt?:
        | string
        | number
        | {
            checkInterval?: number;
            hidden?: boolean;
            time: string
            | number;
            upcoming?: { checkInterval?: number; time: number };
        };
    meta?: {
        duration?: number;
        folderId?: string;
        kinescopeUrl?: string;
        projectId?: string;
        type?: "live"
        | "vod";
        url?: string;
        workspaceId?: string;
    };
    poster?: | string
    | ExcludeTypes<
        Pick<PosterProps, "crossOrigin" | "src" | "src0">,
        "",
        { pick: "crossOrigin" },
    >
    | PosterTimeRange;
    projection?: boolean | "360";
    quality?: VideoQuality;
    qualityLabels?: QualityLabels;
    sources: VideoSources<string>;
    storyboard?: StoryboardOptions;
    subtitle?: string;
    time?: number;
    title?: string;
    vtt?: readonly (
        Readonly<Omit<{}, "kind"> & {}> & Pick<
            TrackHTMLAttributes<HTMLTrackElement>,
            "srcLang",
        >
    )[];
}
